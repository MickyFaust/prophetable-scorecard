const express = require('express'); const fs = require('fs'); const path = require('path'); const app = express(); // --- CONFIGURATION --- const PORT = process.env.PORT || 3000; const PASSWORD = 'Dravrah1!1'; // Your Syndicate Key const DATA_PATH = '/app/data/plays.json'; // Points to your Railway Volume // --- MIDDLEWARE --- app.use(express.json()); app.use(express.urlencoded({ extended: true })); app.use(express.static('public')); // This makes admin.html and index.html work automatically // --- API ROUTES --- // 1. Get all plays for the scorecard app.get('/api/plays', (req, res) => { if (fs.existsSync(DATA_PATH)) { const data = fs.readFileSync(DATA_PATH); res.json(JSON.parse(data)); } else { res.json([]); // Return empty list if no plays yet } }); // 2. Add a new play via the Admin Form app.post('/admin/add-play', (req, res) => { const { password, league, matchup, pick, units } = req.body; // Check Password if (password !== PASSWORD) { return res.status(401).send('Unauthorized: Incorrect Syndicate Key'); } let plays = []; if (fs.existsSync(DATA_PATH)) { plays = JSON.parse(fs.readFileSync(DATA_PATH)); } const newPlay = { id: Date.now(), date: new Date().toLocaleDateString(), league, matchup, pick, units: parseFloat(units) }; plays.push(newPlay); // Ensure the data directory exists const dir = path.dirname(DATA_PATH); if (!fs.existsSync(dir)) { fs.mkdirSync(dir, { recursive: true }); } // Save to the Volume fs.writeFileSync(DATA_PATH, JSON.stringify(plays, null, 2)); // Redirect back to home to see the result res.redirect('/'); }); // --- START SERVER --- app.listen(PORT, () => { console.log(`Syndicate Engine active on port ${PORT}`); });
